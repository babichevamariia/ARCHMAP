<html lang="en">

<head>

  <meta charset="UTF-8">

  <meta name="viewport" content="width=device-width, initial-scale=1.0">

  <title>Digital Map of Italian Architecture Archives</title>

  <link rel="stylesheet" href="https://unpkg.com/leaflet/dist/leaflet.css" />

  <style>

    #map { height: 100vh; }

    .popup-title { font-weight: bold; font-size: 14px; }

    .popup-sub { font-size: 12px; margin-top: 4px; }

    .popup-notes { font-size: 11px; font-style: italic; margin-top: 6px; color: #555; }

  </style>

</head>

<body>

  <div id="map"></div>


  <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

  <script src="https://cdn.jsdelivr.net/npm/papaparse@5.3.2/papaparse.min.js"></script>

  <script>

    // Инициализация карты

    var map = L.map('map').setView([41.8719, 12.5674], 6); // Центр Италии


    // Подложка карты

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

      attribution: '© OpenStreetMap contributors'

    }).addTo(map);


    // Загружаем данные из CSV

    Papa.parse("data.csv", {

      download: true,

      header: true,

      complete: function(results) {

        results.data.forEach(item => {

          if (!item.coords_lat || !item.coords_lng) return;


          var popupContent = `

            <div class="popup-title">${item.project} (${item.year})</div>

            <div class="popup-sub">Архитектор: <a href="architect.html?name=${encodeURIComponent(item.architect)}">${item.architect}</a> (${item.birth_death || ''})</div>

            <div class="popup-sub">Тип проекта: ${item.project_type || ''}</div>

            <div class="popup-sub">Тип документа: ${item.document_type || ''}</div>

            <div class="popup-sub">Институция: <a href="${item.website}" target="_blank">${item.institution}</a></div>

            <div class="popup-sub">Местоположение: ${item.city || ''}, ${item.country || ''}</div>

            ${item.notes ? `<div class="popup-notes">${item.notes}</div>` : ''}

          `;


          L.marker([parseFloat(item.coords_lat), parseFloat(item.coords_lng)])

            .addTo(map)

            .bindPopup(popupContent);

        });

      }

    });

  </script>

</body>

</html>
