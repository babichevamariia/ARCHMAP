  <script src="https://cdn.jsdelivr.net/npm/papaparse@5.3.2/papaparse.min.js"></script>

  <script>

    // Получаем имя архитектора из URL

    const urlParams = new URLSearchParams(window.location.search);

    const architectName = urlParams.get('name');


    document.getElementById('architect-name').textContent = architectName;


    // Загружаем CSV

    Papa.parse("data.csv", {

      download: true,

      header: true,

      complete: function(results) {

        const data = results.data.filter(item => item.architect === architectName);

        if (data.length === 0) {

          document.getElementById('projects').innerHTML = '<p>Нет данных для этого архитектора.</p>';

          return;

        }


        // Биографическая справка (берём годы жизни из первой строки)

        const bio = `${architectName} (${data[0].birth_death || ''})`;

        document.getElementById('bio').textContent = bio;


        // Список проектов

        const projectsDiv = document.getElementById('projects');

        data.forEach(item => {

          const div = document.createElement('div');

          div.className = 'project';

          div.innerHTML = `

            <div class="project-title">${item.project} (${item.year})</div>

            <div>Тип проекта: ${item.project_type || ''}</div>

            <div>Тип документа: ${item.document_type || ''}</div>

            <div>Институция: <a href="${item.website}" target="_blank">${item.institution}</a></div>

            <div>Местоположение: ${item.city || ''}, ${item.country || ''}</div>

            ${item.notes ? `<div class="notes">${item.notes}</div>` : ''}

          `;

          projectsDiv.appendChild(div);

        });

      }

    });

  </script>
